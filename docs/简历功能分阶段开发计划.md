package pkg

import (
    "bytes"
    "fmt"
    "strings"

    "github.com/ledongthuc/pdf"
    "github.com/unidoc/unioffice/document"
)

// ExtractTextFromPDF 从 PDF 文件提取文本
func ExtractTextFromPDF(filePath string) (string, error) {
    f, r, err := pdf.Open(filePath)
    if err != nil {
        return "", fmt.Errorf("failed to open PDF: %w", err)
    }
    defer f.Close()

    var buf bytes.Buffer
    b, err := r.GetPlainText()
    if err != nil {
        return "", fmt.Errorf("failed to get plain text: %w", err)
    }
    buf.ReadFrom(b)

    return buf.String(), nil
}

// ExtractTextFromDOCX 从 DOCX 文件提取文本
func ExtractTextFromDOCX(filePath string) (string, error) {
    doc, err := document.Open(filePath)
    if err != nil {
        return "", fmt.Errorf("failed to open DOCX: %w", err)
    }
    defer doc.Close()

    var text strings.Builder
    for _, para := range doc.Paragraphs() {
        text.WriteString(para.Text())
        text.WriteString("\n")
    }

    return text.String(), nil
}

// ExtractText 根据文件扩展名提取文本
func ExtractText(filePath string) (string, error) {
    ext := strings.ToLower(filePath)
    
    if strings.HasSuffix(ext, ".pdf") {
        return ExtractTextFromPDF(filePath)
    } else if strings.HasSuffix(ext, ".docx") {
        return ExtractTextFromDOCX(filePath)
    }
    
    return "", fmt.Errorf("unsupported file format: %s", ext)
}
```

func (l *UploadResumeLogic) UploadResume(req *types.ResumeUploadReq) (*types.StudentResp, error) {
    // 1. 验证参数
    if req.FileContent == "" || req.FileName == "" {
        return &types.StudentResp{
            Code: errors.CodeInvalidParam,
            Msg:  "fileContent and fileName are required",
        }, nil
    }

    // 2. 验证文件格式
    fileName := strings.ToLower(req.FileName)
    if !strings.HasSuffix(fileName, ".pdf") && !strings.HasSuffix(fileName, ".docx") {
        return &types.StudentResp{
            Code: errors.CodeInvalidParam,
            Msg:  "unsupported file format, only PDF and DOCX are supported",
        }, nil
    }

    // 3. 验证文件大小（base64 编码后大约比原文件大 33%）
    if len(req.FileContent) > 10*1024*1024*4/3 {
        return &types.StudentResp{
            Code: errors.CodeInvalidParam,
            Msg:  "file size exceeds 10MB limit",
        }, nil
    }

    // 4. 解码 Base64
    fileData, err := base64.StdEncoding.DecodeString(req.FileContent)
    if err != nil {
        logx.Errorf("Failed to decode base64: %v", err)
        return &types.StudentResp{
            Code: errors.CodeInvalidParam,
            Msg:  "invalid base64 content",
        }, nil
    }

    // 5. 保存临时文件
    tempFile := fmt.Sprintf("/tmp/resume_%d%s", time.Now().UnixNano(), filepath.Ext(req.FileName))
    if err := os.WriteFile(tempFile, fileData, 0644); err != nil {
        logx.Errorf("Failed to save temp file: %v", err)
        return &types.StudentResp{
            Code: errors.CodeInternalError,
            Msg:  "failed to save file",
        }, nil
    }
    defer os.Remove(tempFile) // 清理临时文件

    // 6. 提取文本内容
    resumeText, err := pkg.ExtractText(tempFile)
    if err != nil {
        logx.Errorf("Failed to extract text: %v", err)
        return &types.StudentResp{
            Code: errors.CodeInternalError,
            Msg:  "failed to extract text from file",
        }, nil
    }

    // 7. 验证提取的文本内容
    if len(strings.TrimSpace(resumeText)) < 50 {
        return &types.StudentResp{
            Code: errors.CodeInvalidParam,
            Msg:  "extracted text is too short, please check your file",
        }, nil
    }

    // 8. 调用 AI API 解析
    aiResult, err := l.svcCtx.AIProvider.GenerateStudentProfile(l.ctx, resumeText)
    if err != nil {
        logx.Errorf("GenerateStudentProfile failed: %v", err)
        return &types.StudentResp{
            Code: errors.CodeInternalError,
            Msg:  "failed to parse resume with AI",
        }, nil
    }

    // 9. 解析 AI 返回的 JSON
    var profile types.StudentProfile
    if err := parseAIResult(aiResult, &profile); err != nil {
        logx.Errorf("Failed to parse AI result: %v", err)
        return &types.StudentResp{
            Code: errors.CodeInternalError,
            Msg:  "failed to parse AI result",
        }, nil
    }

    // 10. 设置元数据
    profile.UserId = getUserId(l.ctx)
    profile.CreatedAt = time.Now().Unix()
    profile.UpdatedAt = time.Now().Unix()

    return &types.StudentResp{
        Code: errors.CodeSuccess,
        Msg:  "success",
        Data: &profile,
    }, nil
}

// parseAIResult 解析 AI 返回的结果
func parseAIResult(aiResult string, profile *types.StudentProfile) error {
    // 提取 JSON（可能包含 Markdown 格式）
    jsonStr := aiResult
    if strings.Contains(aiResult, "```json") {
        start := strings.Index(aiResult, "```json")
        end := strings.LastIndex(aiResult, "```")
        if start != -1 && end != -1 {
            jsonStr = strings.TrimSpace(aiResult[start+7 : end])
        }
    }

    // 解析 JSON
    if err := json.Unmarshal([]byte(jsonStr), profile); err != nil {
        return fmt.Errorf("failed to unmarshal JSON: %w", err)
    }

    return nil
}
```

**文件**：`high-school-worker-design-forend/src/pages/Resume/index.tsx`
```typescript
import { useState } from 'react';
import { Card, Upload, Button, message, Steps, Result, Progress, List, Tag } from 'antd';
import { UploadOutlined, FileTextOutlined, CheckCircleOutlined } from '@ant-design/icons';
import type { UploadFile } from 'antd';
import { studentApi } from '../api';

export default function ResumePage() {
  const [fileList, setFileList] = useState<UploadFile[]>([]);
  const [uploading, setUploading] = useState(false);
  const [parsing, setParsing] = useState(false);
  const [parsed, setParsed] = useState(false);
  const [progress, setProgress] = useState(0);
  const [profile, setProfile] = useState<any>(null);

  // 文件转 base64
  const fileToBase64 = (file: File): Promise<string> => {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.readAsDataURL(file);
      reader.onload = () => {
        const base64 = (reader.result as string).split(',')[1];
        resolve(base64);
      };
      reader.onerror = error => reject(error);
    });
  };

  // 处理文件上传
  const handleUpload = async () => {
    if (fileList.length === 0) {
      message.warning('请先选择简历文件');
      return;
    }

    const file = fileList[0].originFileObj;
    if (!file) {
      message.error('文件读取失败');
      return;
    }

    // 验证文件大小（10MB）
    if (file.size > 10 * 1024 * 1024) {
      message.error('文件大小不能超过 10MB');
      return;
    }

    setUploading(true);
    setProgress(0);

    try {
      // 1. 文件转 base64
      setProgress(30);
      const base64Content = await fileToBase64(file);
      setProgress(60);

      // 2. 调用 API
      setParsing(true);
      const response = await studentApi.uploadResume({
        fileContent: base64Content,
        fileName: file.name,
      });

      setProgress(100);

      // 3. 处理响应
      if (response.code === 0) {
        setProfile(response.data);
        setParsed(true);
        message.success('简历解析完成');
      } else {
        message.error(response.msg || '解析失败');
      }
    } catch (error: any) {
      console.error('Upload error:', error);
      message.error(error.response?.data?.msg || '上传失败，请重试');
    } finally {
      setUploading(false);
      setParsing(false);
    }
  };

  // 重新上传
  const handleReset = () => {
    setFileList([]);
    setParsed(false);
    setProfile(null);
    setProgress(0);
  };

  return (
    <div className="min-h-screen bg-gray-50 p-4">
      <Steps
        current={parsed ? 2 : parsing ? 1 : 0}
        className="mb-6"
        items={[
          { title: '上传简历', icon: <UploadOutlined /> },
          { title: 'AI 解析', icon: <FileTextOutlined /> },
          { title: '优化建议', icon: <CheckCircleOutlined /> },
        ]}
      />

      {!parsed ? (
        <Card title="上传简历">
          <Upload
            fileList={fileList}
            onChange={({ fileList }) => setFileList(fileList)}
            beforeUpload={() => false}
            accept=".pdf,.doc,.docx"
            maxCount={1}
          >
            <Button icon={<UploadOutlined />}>点击选择文件</Button>
          </Upload>
          <p className="text-gray-500 text-sm mt-2">
            支持 PDF、Word 格式，文件大小不超过 10MB
          </p>

          {progress > 0 && progress < 100 && (
            <Progress percent={progress} className="mt-4" />
          )}

          <Button
            type="primary"
            block
            className="mt-4"
            onClick={handleUpload}
            loading={uploading || parsing}
            disabled={fileList.length === 0}
          >
            {parsing ? 'AI 解析中...' : uploading ? '上传中...' : '开始解析'}
          </Button>
        </Card>
      ) : (
        <Result
          status="success"
          title="简历解析完成"
          subTitle="AI 已完成简历分析，以下是详细信息和优化建议"
          extra={[
            <Button type="primary" key="optimize">查看优化建议</Button>,
            <Button key="compare">双版本对比</Button>,
            <Button key="reset" onClick={handleReset}>重新上传</Button>,
          ]}
        >
          <div className="text-left space-y-4">
            {/* 基础信息 */}
            <Card title="基础信息" size="small">
              <p><strong>姓名：</strong>{profile?.name || '未提取'}</p>
              <p><strong>学历：</strong>{profile?.education || '未提取'}</p>
              <p><strong>专业：</strong>{profile?.major || '未提取'}</p>
              <p><strong>毕业年份：</strong>{profile?.graduationYear || '未提取'}</p>
            </Card>

            {/* 技能列表 */}
            <Card title="技能列表" size="small">
              <List
                dataSource={profile?.skills || []}
                renderItem={(skill: any) => (
                  <List.Item>
                    <Tag color="blue">{skill.name}</Tag>
                    <span className="ml-2">
                      等级: {skill.level}/100 | 掌握年限: {skill.years} 年
                    </span>
                  </List.Item>
                )}
              />
            </Card>

            {/* 证书列表 */}
            <Card title="证书列表" size="small">
              <List
                dataSource={profile?.certificates || []}
                renderItem={(cert: any) => (
                  <List.Item>
                    <Tag color="green">{cert.name}</Tag>
                    <span className="ml-2">
                      等级: {cert.level} | 获得年份: {cert.year}
                    </span>
                  </List.Item>
                )}
              />
            </Card>

            {/* 完整度和竞争力 */}

            </Card>

            {/* 优化建议 */}
            <Card title="优化建议" size="small">
              <List
                dataSource={profile?.suggestions || []}

                    <span>• {suggestion}</span>
                  </List.Item>
                )}
              />
            </Card>
          </div>

      )}
    </div>
  );
}

    skillsJSON, _ := json.Marshal(profile.Skills)
    certsJSON, _ := json.Marshal(profile.Certificates)
    internshipJSON, _ := json.Marshal(profile.Internship)
    projectsJSON, _ := json.Marshal(profile.Projects)

    // 11. 查询用户是否已有档案
    existing, err := l.svcCtx.StudentModel.FindByUserId(l.ctx, profile.UserId)

            Projects:        string(projectsJSON),
            ResumeContent:   resumeText,

            Competitiveness: profile.Competitiveness,
            UpdatedAt:       time.Now().Unix(),
        }
