### 1.3 缺少JWT认证中间件

**问题描述**：除了login/register外，所有接口都需要携带token认证，但当前没有实现

**建议修改**：在 `internal/middleware/middleware.go` 中添加认证中间件

```go
package middleware

import (
    "context"
    "net/http"
    "strings"

    "career-api/common/errors"
    "career-api/internal/pkg"

    "github.com/zeromicro/go-zero/rest/httpx"
)

type AuthMiddleware struct {
    accessSecret string
}

func NewAuthMiddleware(accessSecret string) *AuthMiddleware {
    return &AuthMiddleware{accessSecret: accessSecret}
}

func (m *AuthMiddleware) Handle(next http.HandlerFunc) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        // 排除不需要认证的路径
        path := r.URL.Path
        if strings.Contains(path, "/user/login") ||
           strings.Contains(path, "/user/register") ||
           strings.Contains(path, "/health") {
            next(w, r)
            return
        }

        // 获取token
        auth := r.Header.Get("Authorization")
        if auth == "" {
            httpx.Error(w, errors.NewCodeError(errors.CodeUnauthorized, "missing authorization header"))
            return
        }

        // 验证Bearer token格式
        parts := strings.SplitN(auth, " ", 2)
        if len(parts) != 2 || parts[0] != "Bearer" {
            httpx.Error(w, errors.NewCodeError(errors.CodeUnauthorized, "invalid authorization format"))
            return
        }

        // 验证JWT token
        claims, err := pkg.ParseToken(parts[1], m.accessSecret)
        if err != nil {
            httpx.Error(w, errors.NewCodeError(errors.CodeUnauthorized, "invalid token"))
            return
        }

        // 将用户信息存入context
        ctx := context.WithValue(r.Context(), "userId", claims.UserId)
        ctx = context.WithValue(ctx, "username", claims.Username)
        ctx = context.WithValue(ctx, "role", claims.Role)

        next(w, r.WithContext(ctx))
    }
}
```

```go
type GenerateReportStreamLogic struct {

func NewGenerateReportStreamLogic(ctx context.Context, svcCtx *svc.ServiceContext) *GenerateReportStreamLogic {
    return &GenerateReportStreamLogic{
        ctx:    ctx,
        svcCtx: svcCtx,
    }
}

func (l *GenerateReportStreamLogic) GenerateReportStream(w http.ResponseWriter, req *types.GenerateReportStreamReq) {
    // 设置SSE响应头
    w.Header().Set("Content-Type", "text/event-stream")
    w.Header().Set("Cache-Control", "no-cache")
    w.Header().Set("Connection", "keep-alive")
    w.Header().Set("Access-Control-Allow-Origin", "*")

    // 创建flusher用于实时推送
    flusher, ok := w.(http.Flusher)
    if !ok {
        http.Error(w, "SSE not supported", http.StatusInternalServerError)
        return
    }

    // 获取用户ID
    userId, ok := l.ctx.Value("userId").(int64)
    if !ok {
        l.sendSSEEvent(w, flusher, "error", map[string]interface{}{
            "code": errors.CodeUnauthorized,
            "msg":  "unauthorized",
        })
        return
    }

    // 获取学生信息
    student, err := l.svcCtx.StudentModel.FindOneByUserId(l.ctx, userId)
    if err != nil {
        l.sendSSEEvent(w, flusher, "error", map[string]interface{}{
            "code": errors.CodeNotFound,
            "msg":  "student profile not found",
        })
        return
    }

    // 发送开始事件
    l.sendSSEEvent(w, flusher, "status", map[string]interface{}{
        "type":    "start",
        "message": "开始生成职业规划报告...",
    })

    // 生成报告内容
    content, err := l.svcCtx.AIProvider.GenerateCareerReport(l.ctx, ai.ReportGenerationRequest{
        StudentProfile: fmt.Sprintf("Name: %s, Education: %s, Major: %s",
            student.Name, student.Education, student.Major),
        JobProfile: fmt.Sprintf("Track: %s", req.Track),
        MatchResult: "AI analysis",
        Options: ai.ReportOptions{
            IncludeGapAnalysis: true,
            IncludeActionPlan:  true,
            DetailedLevel:      2,
        },
    })

    if err != nil {
        logx.Errorf("GenerateCareerReport failed: %v", err)
        l.sendSSEEvent(w, flusher, "error", map[string]interface{}{
            "code": errors.CodeInternalError,
            "msg":  "failed to generate report",
        })
        return
    }

    // 模拟流式输出
    buffer := ""
    for i, char := range content {
        buffer += string(char)
        if len(buffer) >= 50 || i == len(content)-1 {
            l.sendSSEEvent(w, flusher, "content", map[string]interface{}{
                "type":    "text",
                "content": buffer,
            })
            buffer = ""
            time.Sleep(20 * time.Millisecond) // 模拟打字效果
        }
    }

    // 发送完成事件
    l.sendSSEEvent(w, flusher, "done", map[string]interface{}{
        "type":    "complete",
        "message": "报告生成完成",
    })
}

func (l *GenerateReportStreamLogic) sendSSEEvent(w io.Writer, flusher http.Flusher, eventType string, data interface{}) {
    jsonData, _ := json.Marshal(data)
    fmt.Fprintf(w, "event: %s\n", eventType)
    fmt.Fprintf(w, "data: %s\n\n", jsonData)
    flusher.Flush()
}
```

### 3.1 密码明文存储问题

**问题描述**：当前密码以明文形式存储，存在严重安全隐患

**建议创建**：`internal/pkg/password.go`

```go
package pkg

import "golang.org/x/crypto/bcrypt"

// HashPassword 对密码进行哈希加密
func HashPassword(password string) (string, error) {
    bytes, err := bcrypt.GenerateFromPassword([]byte(password), bcrypt.DefaultCost)
    return string(bytes), err
}

// CheckPassword 验证密码
func CheckPassword(password, hash string) bool {
    err := bcrypt.CompareHashAndPassword([]byte(hash), []byte(password))
    return err == nil
}
```

### 3.2 缺少JWT Token实现

**建议创建**：`internal/pkg/jwt.go`

```go
package pkg

import (
    "errors"
    "time"

    "github.com/golang-jwt/jwt/v5"
)

// Claims JWT声明
type Claims struct {
    UserId   int64  `json:"userId"`
    Username string `json:"username"`
    Role     string `json:"role"`
    jwt.RegisteredClaims
}

// GenerateToken 生成JWT token
func GenerateToken(userId int64, username, role, secret string, expireSeconds int64) (string, error) {
    claims := Claims{
        UserId:   userId,
        Username: username,
        Role:     role,
        RegisteredClaims: jwt.RegisteredClaims{
            ExpiresAt: jwt.NewNumericDate(time.Now().Add(time.Duration(expireSeconds) * time.Second)),
            IssuedAt:  jwt.NewNumericDate(time.Now()),
            NotBefore: jwt.NewNumericDate(time.Now()),
        },
    }

    token := jwt.NewWithClaims(jwt.SigningMethodHS256, claims)
    return token.SignedString([]byte(secret))
}

// ParseToken 解析JWT token
func ParseToken(tokenString, secret string) (*Claims, error) {
    token, err := jwt.ParseWithClaims(tokenString, &Claims{}, func(token *jwt.Token) (interface{}, error) {
        return []byte(secret), nil
    })

    if err != nil {
        return nil, err
    }

    if claims, ok := token.Claims.(*Claims); ok && token.Valid {
        return claims, nil
    }

    return nil, errors.New("invalid token")
}
```

**在 `internal/logic/userlogic.go` 中使用**：

```go
func (l *LoginLogic) Login(req *types.LoginReq) (*types.LoginResp, error) {

        user.Username,

        l.svcCtx.Config.Auth.AccessSecret,

        return &types.LoginResp{
            Code: errors.CodeInternalError,

        }, err
    }

    return &types.LoginResp{
        Code:   errors.CodeSuccess,
        Msg:    "success",
        Token:  token,
```

```go
// GetStudentWithCache 带缓存的学生信息查询
func (s *ServiceContext) GetStudentWithCache(ctx context.Context, studentId int64) (*model.Students, error) {
    cacheKey := fmt.Sprintf("student:%d", studentId)

    // 先从Redis获取
    cached, err := s.Redis.Get(cacheKey)
    if err == nil && cached != "" {
        var student model.Students
        if err := json.Unmarshal([]byte(cached), &student); err == nil {
            return &student, nil
        }
    }

    // 缓存未命中，从数据库获取
    student, err := s.StudentModel.FindOne(ctx, studentId)
    if err != nil {
        return nil, err
    }

    // 写入缓存（30分钟过期）
    studentJSON, _ := json.Marshal(student)
    s.Redis.Setex(cacheKey, string(studentJSON), 1800)

    return student, nil
}

// ClearStudentCache 清除学生缓存
func (s *ServiceContext) ClearStudentCache(studentId int64) error {
    cacheKey := fmt.Sprintf("student:%d", studentId)
    return s.Redis.Del(cacheKey)
}
```

### 4.4 数据库连接池优化

```go
import (
    "time"

    "github.com/zeromicro/go-zero/core/stores/sqlx"
)

func NewServiceContext(c *config.Config) *ServiceContext {
    // 优化数据库连接池配置
    mysqlConn := sqlx.NewMysql(c.Mysql.DataSource,
        sqlx.WithMaxConns(c.Mysql.MaxOpenConns),
        sqlx.WithMaxIdleConns(c.Mysql.MaxIdleConns),
        sqlx.WithConnMaxLifetime(time.Duration(c.Mysql.ConnMaxLifetime)*time.Second),
        sqlx.WithConnMaxIdleTime(time.Duration(1800)*time.Second), // 30分钟空闲超时
    )

    redisClient := redis.New(c.Redis.Host, redis.WithPass(c.Redis.Pass))

    aiProvider := ai.NewOpenAIProvider(
        c.AI.ApiKey,
        c.AI.Model,
        c.AI.BaseURL,
        time.Duration(c.AI.Timeout)*time.Second,
    )

    return &ServiceContext{
        Config:             c,
        Redis:              redisClient,
        DB:                 mysqlConn,
        AIProvider:         aiProvider,
        AITimeout:          time.Duration(c.AI.Timeout) * time.Second,
        UserModel:          model.NewUsersModel(mysqlConn),
        JobModel:           model.NewJobsModel(mysqlConn),
        StudentModel:       model.NewStudentsModel(mysqlConn),
        ReportModel:        model.NewCareerReportsModel(mysqlConn),
        MatchModel:         model.NewMatchRecordsModel(mysqlConn),

    }
}
```
