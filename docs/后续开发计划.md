```go
package pkg

import "golang.org/x/crypto/bcrypt"

// HashPassword 对密码进行哈希加密
func HashPassword(password string) (string, error) {
    bytes, err := bcrypt.GenerateFromPassword([]byte(password), bcrypt.DefaultCost)
    return string(bytes), err
}

// CheckPassword 验证密码
func CheckPassword(password, hash string) bool {
    err := bcrypt.CompareHashAndPassword([]byte(hash), []byte(password))
    return err == nil
}
```

```go
// GetStudentWithCache 带缓存的学生信息查询
func (s *ServiceContext) GetStudentWithCache(ctx context.Context, studentId int64) (*model.Students, error) {
    cacheKey := fmt.Sprintf("student:%d", studentId)

    // 先从Redis获取
    cached, err := s.Redis.Get(cacheKey)
    if err == nil && cached != "" {
        var student model.Students
        if err := json.Unmarshal([]byte(cached), &student); err == nil {
            return &student, nil
        }
    }

    // 缓存未命中，从数据库获取
    student, err := s.StudentModel.FindOne(ctx, studentId)
    if err != nil {
        return nil, err
    }

    // 写入缓存（30分钟过期）
    studentJSON, _ := json.Marshal(student)
    s.Redis.Setex(cacheKey, string(studentJSON), 1800)

    return student, nil
}

// ClearStudentCache 清除学生缓存
func (s *ServiceContext) ClearStudentCache(studentId int64) error {
    cacheKey := fmt.Sprintf("student:%d", studentId)
    return s.Redis.Del(cacheKey)
}
```

```go
server.AddRoutes(
    []rest.Route{
        {
            Method:  http.MethodPost,
            Path:    "/api/v1/reports/generate",
            Handler: generateReportHandler(serverCtx),
        },
        {
            Method:  http.MethodPost,
            Path:    "/api/v1/reports/generate-stream",
            Handler: generateReportStreamHandler(serverCtx),
        },
        {
            Method:  http.MethodPost,
            Path:    "/api/v1/students/resume",
            Handler: uploadResumeHandler(serverCtx),
        },
    },
    rest.WithTimeout(60*time.Second),
    rest.WithRateLimit(&rest.RateLimit{
        Period:  time.Second,
        Quota:   serverCtx.Config.RateLimit.TokensPerSecond,
        Burst:   serverCtx.Config.RateLimit.Burst,
    }),
    rest.WithCircuitBreaker(rest.CircuitBreaker{
        Enable:    !serverCtx.Config.CircuitBreaker.ForceOpen,
        Sleep:     time.Duration(serverCtx.Config.CircuitBreaker.SleepWindow) * time.Millisecond,
        Threshold: serverCtx.Config.CircuitBreaker.ErrorPercentThreshold,
    }),
)
```

#### 配置文件优化
```yaml
RateLimit:
  TokensPerSecond: 100  # 每秒令牌数
  Burst: 200            # 突发令牌数

  ForceOpen: false
  SleepWindow: 5000              # 熔断休眠窗口（毫秒）
  ErrorPercentThreshold: 50      # 错误百分比阈值
```
