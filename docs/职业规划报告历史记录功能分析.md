### 后端实现情况

#### ✅ 已完成的部分

**1. 数据库设计完善**
- `career_reports` 表结构完整，包含所有必要字段：
  - `student_id`：学生ID
  - `target_job_id`：目标岗位ID
  - `title`：报告标题
  - `content`：JSON格式的内容（技能、时间轴、完整度等）
  - `overview`：概览信息（JSON）
  - `match_analysis`：匹配分析（JSON）
  - `career_path`：职业路径（JSON）
  - `action_plan`：行动计划（JSON）
  - `status`：状态（draft/completed/failed）
  - `created_at`、`updated_at`：时间戳

**2. 报告生成和保存**
- `GenerateReportStreamLogic` 已实现AI报告生成
- `saveReportToDatabase` 方法正确保存报告到数据库
- 使用JSON格式存储以下数据：
  - `skills`：技能列表（名称、等级、状态）
  - `timeline`：时间轴（日期、标题、描述）
  - `completeness`：完整度（0-100）
  - `competitiveness`：竞争力指数（0-100）

**3. Model层支持**
- `CareerReportsModel.FindAll()`：支持分页查询和过滤
  - 参数：`page`（页码）、`pageSize`（每页数量）、`studentId`（学生ID）、`status`（状态）

- `CareerReportsModel.FindOne()`：查询单个报告详情
- `CareerReportsModel.Insert()`：插入报告记录（自动设置时间戳）

**4. API接口定义**
- `GET /api/v1/reports/generate-stream`：流式生成报告（SSE）
- `GET /api/v1/reports/me`：获取我的报告列表（待实现）
- `GET /api/v1/reports/:id`：获取单个报告详情
- `GET /api/v1/reports`：分页查询报告列表
- `PUT /api/v1/reports`：更新报告
- `DELETE /api/v1/reports/:id`：删除报告

```go
package report

import (
	"context"

	"career-api/common/errors"
	"career-api/internal/ctxkeys"
	"career-api/internal/svc"
	"career-api/internal/types"

	"github.com/zeromicro/go-zero/core/logx"
)

type GetMyReportsLogic struct {
	logx.Logger
	ctx    context.Context
	svcCtx *svc.ServiceContext
}

func NewGetMyReportsLogic(ctx context.Context, svcCtx *svc.ServiceContext) *GetMyReportsLogic {
	return &GetMyReportsLogic{
		Logger: logx.WithContext(ctx),
		ctx:    ctx,
		svcCtx: svcCtx,
	}
}

func (l *GetMyReportsLogic) GetMyReports() (*types.ReportListResultResp, error) {
	// 从上下文获取 userId
	userId, ok := l.ctx.Value(ctxkeys.UserIDKey).(int64)
	if !ok {
		return &types.ReportListResultResp{
			Code: errors.CodeUnauthorized,
			Msg:  "unauthorized",
		}, nil
	}

	// 获取分页参数（默认值）
	page := 1
	pageSize := 10

	// 查询该用户的报告列表
	reports, total, err := l.svcCtx.ReportModel.FindAll(
		l.ctx,
		page,
		pageSize,
		userId,
		"completed", // 只返回已完成的报告
	)
	if err != nil {
		logx.Errorf("GetMyReports failed: %v", err)
		return &types.ReportListResultResp{
			Code: errors.CodeInternalError,
			Msg:  "failed to get reports",
		}, err
	}

	// 转换为响应格式
	list := make([]types.CareerReport, 0, len(reports))
	for _, report := range reports {
		// 解析content JSON获取预览信息
		var preview struct {
			Completeness   float64 `json:"completeness"`
			Competitiveness float64 `json:"competitiveness"`
		}
		if report.Content.Valid {
			json.Unmarshal([]byte(report.Content.String), &preview)
		}

		list = append(list, types.CareerReport{
			Id:        report.Id,
			StudentId: report.StudentId,
			Title:     report.Title.String,
			Content:   report.Content.String,
			Status:    report.Status,
			CreatedAt: report.CreatedAt,
			UpdatedAt: report.UpdatedAt,
		})
	}

	return &types.ReportListResultResp{
		Code: 0,
		Msg:  "success",
		Data: &types.ReportListResp{
			Total: total,
			List:  list,
		},
	}, nil
}
```

func (l *GenerateReportStreamLogic) saveReportToDatabase(student *model.Students, content ReportContent, track string) {
	contentJson, _ := json.Marshal(content)

	reportType := "full"
	if track == "quick" {
		reportType = "quick"
	} else if track == "gap" {
		reportType = "gap"
	}

	_, err := l.svcCtx.ReportModel.Insert(l.ctx, &model.CareerReports{
		StudentId: student.UserId,
		Title:     sql.NullString{String: fmt.Sprintf("职业规划报告 - %s", track), Valid: true},
		Content:   sql.NullString{String: string(contentJson), Valid: true},
		Status:    "completed",
		CreatedAt: time.Now().Unix(),
		UpdatedAt: time.Now().Unix(),
	})

	if err != nil {
		logx.Errorf("保存报告失败: %v", err)
	}
}
